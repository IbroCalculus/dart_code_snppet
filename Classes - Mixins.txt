A mixin is a class that contains methods and fields that can be used by other classes.
Mixins are used to add functionality to a class without using inheritance, allowing for code reuse in a more flexible and modular way.


class Aclass{
  void methodA(){
    print("This is a method from class A");
  }
}


mixin Bclass{							//NB: mixin
  void methodB(){
    print("This is a method from class B");
  }
}


class Cclass extends Aclass with Bclass, Dclass{		// It's usage		
  void methodC(){
    print("This is a method from class C");
  }
}

mixin Dclass{							//NB: mixin
  void methodD(){
    print("This is a method from class D");
  }
}

------------------------

  Cclass cclass1 = Cclass();
  cclass1.methodA();
  cclass1.methodB();
  cclass1.methodC();


=============================== EXAMPLE 2 ============================

mixin Walk{
  void walk(){}
}

mixin Talk{
  void talk(){}
}

mixin Reproduce{
  void reproduce(){}
}

class Human with Talk, Walk, Reproduce{
 void description(){
   print("This is a human");
 } 
 
 @override
  void talk() {
   print("Humans can talk");
  }
 
}

class Animal with Walk, Reproduce{
 void description(){
   print("This is an animal");
 } 
}


void main(){  
  Animal animal = Animal();
  Human human = Human();
  
  human.talk();
  // animal.talk();       // .talk() not available, because the mixin not used in class Animal
}
