
void main() {
  
  print(MathUtils.PI);
  print(MathUtils.add(34, 16));
  
  MathUtils mathUtils = new MathUtils();
  print(mathUtils.subtract(34, 16));
  
}



class MathUtils {
  
  static const PI = 3.14;
  static const E = 2.71;  
  
  static int add(int a, int b) {
    return a + b;
  }
  
  int subtract(int a, int b) {        // Non static method
    return a - b;
  }
  
  
  
  
}
